    <script>
        const channels = [
            { name: 'Welcome', file: null },
            { name: 'Tries', file: 'trie.html' },
            { name: 'Heaps', file: 'heap.html' },
            { name: 'Hash Tables', file: 'hashtable.html' },
            { name: 'Linked Lists', file: 'linkedlist.html' }
        ];

        let currentChannel = 0;
        let visualizationActive = false;

        function updateChannelDisplay() {
            // Hide all channel views
            document.querySelectorAll('.channel-view').forEach(view => {
                view.classList.remove('active');
            });

            // Show current channel
            const currentView = document.querySelector(`[data-channel="${currentChannel}"]`);
            if (currentView) {
                currentView.classList.add('active');
            }

            // Update indicators
            document.querySelectorAll('.indicator-dot').forEach((dot, index) => {
                dot.classList.toggle('active', index === currentChannel);
            });

            // Update buttons
            updateButtonStates();
        }

        function updateButtonStates() {
            const prevBtn = document.getElementById('prevBtn');
            const nextBtn = document.getElementById('nextBtn');
            const startBtn = document.getElementById('startBtn');
            const instructionText = document.getElementById('instructionText');

            if (visualizationActive) {
                prevBtn.disabled = true;
                nextBtn.disabled = true;
                startBtn.textContent = 'Close';
                startBtn.classList.remove('primary');
                instructionText.textContent = 'Press Close to return to channels';
            } else {
                prevBtn.disabled = currentChannel === 0;
                nextBtn.disabled = currentChannel === channels.length - 1;
                
                if (currentChannel === 0) {
                    startBtn.textContent = 'Start';
                    startBtn.classList.add('primary');
                    instructionText.textContent = 'Use arrows to browse channels ‚Ä¢ Press Start to begin';
                } else {
                    startBtn.textContent = 'Start';
                    startBtn.classList.add('primary');
                    instructionText.textContent = `Launch ${channels[currentChannel].name} visualization`;
                }
            }
        }

        function previousChannel() {
            if (currentChannel > 0 && !visualizationActive) {
                currentChannel--;
                updateChannelDisplay();
            }
        }

        function nextChannel() {
            if (currentChannel < channels.length - 1 && !visualizationActive) {
                currentChannel++;
                updateChannelDisplay();
            }
        }

        function startVisualization() {
            if (visualizationActive) {
                // Close visualization
                closeVisualization();
            } else if (currentChannel > 0) {
                // Launch visualization
                const channel = channels[currentChannel];
                const iframeView = document.getElementById('iframeView');
                
                // Create iframe
                const iframe = document.createElement('iframe');
                iframe.src = `visualizations/${channel.file}`;
                iframeView.innerHTML = '';
                iframeView.appendChild(iframe);
                
                // Show iframe view
                iframeView.classList.add('active');
                visualizationActive = true;
                updateButtonStates();
            } else {
                // From welcome screen, go to first channel
                currentChannel = 1;
                updateChannelDisplay();
            }
        }

        function closeVisualization() {
            const iframeView = document.getElementById('iframeView');
            iframeView.classList.remove('active');
            iframeView.innerHTML = '';
            visualizationActive = false;
            updateButtonStates();
        }

        function returnToMenu() {
            if (visualizationActive) {
                closeVisualization();
            }
            currentChannel = 0;
            updateChannelDisplay();
        }

        // Keyboard controls
        document.addEventListener('keydown', function(e) {
            if (e.key === 'ArrowLeft') {
                e.preventDefault();
                previousChannel();
            } else if (e.key === 'ArrowRight') {
                e.preventDefault();
                nextChannel();
            } else if (e.key === 'Enter' || e.key === ' ') {
                e.preventDefault();
                startVisualization();
            } else if (e.key === 'Escape') {
                e.preventDefault();
                if (visualizationActive) {
                    closeVisualization();
                } else {
                    returnToMenu();
                }
            }
        });

        // Initialize
        updateChannelDisplay();
    </script>
